044/** 
045 * Generates events for a WDM network carrying lightpaths in a fixed or flexi grid of wavelengths
046 * 
047 * <p>The design follows the assumptions described in {@link com.net2plan.libraries.WDMUtils WDMUtils} Net2Plan library</p>
048 * <p>The events generated targeted to the event processor module (e.g. {@link com.net2plan.examples.general.onlineSim.Online_evProc_wdm Online_evProc_wdm}) are:</p> 
049 * <ul>
050 * <li>WDMUtils.LightpathAdd: Request to add a new lightpath to the network. In this case, the lightpath is associated to an existing demand, which generates lightpath requests. 
051 * The line rates of the requested lightpaths are chosen among a user-defined set of possible line rates.</li>
052 * <li>WDMUtils.LightpathRemove: Removes the corresponding lightpath, releasing the resources. This occurs if the generator is configured to generate on-demand lightpath connection requests, and the increemntal mode is NOT activated (in the incremental model, lightpaths are setup, and necever released, so traffic always increments).</li>
053 * <li>SimEvent.DemandModify: Modifies the offered traffic of a demand (here a demand is a source that generates lightpath requests).</li>
054 * <li>SimEvent.NodesAndLinksChangeFailureState: Sends these events to the processor, representing network failures and repairs to react to.</li>
055 * </ul>
056 * The average rate of lightpath requests sent can change along time using fast and/or slow fluctuations. Fast fluctuations are 
057 * intended to reflect typical short time-scale traffic variations, while slow fluctuation are more suitable for representing 
058 * multihour (slow) traffic fluctuations (e.g. night vs day traffic). 
059 *  
060 *  In the long-run simulations, lightpath requests and removal are sent. This is used to simulate the operation of a lightpath-on-demand service. 
061 *  With the incremental model, lightpaths are never released, and the traffic only increases. This can be used e.g. in studies that search for the moment in 
062 *  which the network needs an upgrade, since its capacity is exhausted.
063 *  
064 * See the technology conventions used in Net2Plan built-in algorithms and libraries to represent WDM networks. 
065 * @net2plan.keywords WDM, Network recovery: protection, Network recovery: restoration, Multihour optimization
066 * @net2plan.inputParameters 
067 * @author Pablo Pavon-Marino, Jose-Luis Izquierdo-Zaragoza
068 */
069public class Online_evGen_wdm extends IEventGenerator
070{

073        private InputParameter _fail_failureModel = new InputParameter ("_fail_failureModel", "#select# perBidirectionalLinkBundle none SRGfromNetPlan perNode perLink perDirectionalLinkBundle" , "Failure model selection: SRGfromNetPlan, perNode, perLink, perDirectionalLinkBundle, perBidirectionalLinkBundle");
074        private InputParameter _tfFast_fluctuationType = new InputParameter ("_tfFast_fluctuationType", "#select# none random-truncated-gaussian" , "");
075        private InputParameter _trafficType = new InputParameter ("_trafficType", "#select# connection-based-longrun connection-based-incremental " , "");
076        private InputParameter _tfSlow_fluctuationType = new InputParameter ("_tfSlow_fluctuationType", "#select# none time-zone-based" , "");
077        private InputParameter cac_arrivalsPattern = new InputParameter ("cac_arrivalsPattern", "#select# deterministic random-exponential-arrivals-deterministic-duration random-exponential-arrivals-and-duration" , "");
078        private InputParameter trafficLayerId = new InputParameter ("trafficLayerId", (long) -1 , "Layer containing traffic demands (-1 means default layer)");
079        private InputParameter randomSeed = new InputParameter ("randomSeed", (long) 1 , "Seed for the random generator (-1 means random)");
080        private InputParameter cac_avHoldingTimeHours = new InputParameter ("cac_avHoldingTimeHours", (double) 1 , "Default average connection duration (in seconds)" , 0 , false , Double.MAX_VALUE , true);
081        private InputParameter tfFast_timeBetweenDemandFluctuationsHours = new InputParameter ("tfFast_timeBetweenDemandFluctuationsHours", (double) 0.1 , "Average time between two changes of demand offered traffic in a demand (demands behave independently)" , 0 , false , Double.MAX_VALUE , true);
082        private InputParameter tfFast_fluctuationCoefficientOfVariation = new InputParameter ("tfFast_fluctuationCoefficientOfVariation", (double) 1.0 , "Average time between two changes of demand offered traffic in a demand (demands behave independently)" , 0 , false , Double.MAX_VALUE , true);
083        private InputParameter tfFast_maximumFluctuationRelativeFactor = new InputParameter ("tfFast_maximumFluctuationRelativeFactor", (double) 1.0 , "The fluctuation of a demand cannot exceed this percentage from the media" , 0 , true , Double.MAX_VALUE , true);
084        private InputParameter tfSlow_startDate = new InputParameter ("tfSlow_startDate", new SimpleDateFormat(DATE_FORMAT).format(Calendar.getInstance().getTime()) , "Initial date and time of the simulation");
085        private InputParameter tfSlow_timeBetweenDemandFluctuationsHours = new InputParameter ("tfSlow_timeBetweenDemandFluctuationsHours", (double) 1.0 , "Average time between two changes of demand offered traffic in a demand (demands behave independently)" , 0 , false , Double.MAX_VALUE , true);
086        private InputParameter tfSlow_defaultTimezone = new InputParameter ("tfSlow_defaultTimezone", (int) 0 , "Default timezone with respect to UTC (in range [-12, 12])" , -12 , 12);
087        private InputParameter fail_defaultMTTFInHours = new InputParameter ("fail_defaultMTTFInHours", (double) 10 , "Default value for Mean Time To Fail (hours) (unused when failureModel=SRGfromNetPlan)" , 0 , false , Double.MAX_VALUE , true);
088        private InputParameter fail_defaultMTTRInHours = new InputParameter ("fail_defaultMTTRInHours", (double) 12 , "Default value for Mean Time To Repair (hours) (unused when failureModel=SRGfromNetPlan)" , 0 , false , Double.MAX_VALUE , true);
089        private InputParameter fail_statisticalPattern = new InputParameter ("fail_statisticalPattern", "#select# exponential-iid" , "Type of failure and repair statistical pattern");
090        private InputParameter lineRatesPerLightpath_Gbps = new InputParameter ("lineRatesPerLightpath_Gbps", "40 0.5 ; 100 0.5" , "Pairs of the form line-rate-Gbps SPACE probability, where probability stands for the chances of requesting a lightpath of such rate. Pairs are separated among them by character \";\" ");

093        private Random rng;

100        private DoubleMatrix1D slowChangingOfferedTraffic_d; // the offered traffic is the sum of the two

103        private double tfSlow_simTimeOfLastCalendarUpdate;

125        @Override
126        public void initialize(NetPlan initialNetPlan, Map<String, String> algorithmParameters, Map<String, String> simulationParameters, Map<String, String> net2planParameters)
127        {
128                /* Initialize all InputParameter objects defined in this object (this uses Java reflection) */
129                InputParameter.initializeAllInputParameterFieldsOfObject(this, algorithmParameters);
130
131                NetworkLayer trafficLayer = trafficLayerId.getLong () == -1? initialNetPlan.getNetworkLayerDefault () : initialNetPlan.getNetworkLayerFromId(trafficLayerId.getLong ());
132                if (trafficLayer == null) throw new Net2PlanException ("Unknown layer id");
133                final int D = initialNetPlan.getNumberOfDemands(trafficLayer);
134                final int N = initialNetPlan.getNumberOfNodes ();
135                if (D == 0) throw new Net2PlanException("No demands were defined in the original design");
136
137                /* Initialize the transponders information: the line rates of the lightpaths that will be requested */
138                final String [] lineRateInfo_t = StringUtils.split(lineRatesPerLightpath_Gbps.getString() , ";");
139                final int T = lineRateInfo_t.length;
140                if (T == 0) throw new Net2PlanException ("The number of line rates defined cannot be zero");
141                this.accumProbabilitiesLineRates_t = new double [T];
142                this.lineRatesGbps_t = new double [T];
143                this.probabilitiesLineRates_t = new double [T];
144                double sumProbabilities = 0;
145                for (int t = 0; t < T ; t ++)
146                {
147                        double [] vals = StringUtils.toDoubleArray(StringUtils.split(lineRateInfo_t [t]));
148                        this.lineRatesGbps_t [t] = vals [0]; if (lineRatesGbps_t [t] <= 0) throw new Net2PlanException ("The line rate of a lightpath must be positive");
149                        this.probabilitiesLineRates_t [t] = vals [1]; if (vals [1] < 0) throw new Net2PlanException ("Occurrence probablities of the line rates cannot be negative");
150                        sumProbabilities += vals [1];
151                }
152                if (sumProbabilities == 0) Arrays.fill(probabilitiesLineRates_t , 1.0/T);
153                else for (int t = 0; t < T ; t ++) probabilitiesLineRates_t [t] /= sumProbabilities;
154                for (int t = 0 ; t < T ; t ++) accumProbabilitiesLineRates_t [t] = t == 0? probabilitiesLineRates_t [0] : probabilitiesLineRates_t [t] + accumProbabilitiesLineRates_t [t-1];
155                if (Math.abs(accumProbabilitiesLineRates_t [T-1] - 1) > 1e-3) throw new RuntimeException ("Bad");
156                
157                /* More initializations */
158                if (randomSeed.getLong () == -1) randomSeed.initialize((long) RandomUtils.random(0, Long.MAX_VALUE - 1));
159                this.rng = new Random(randomSeed.getLong ());
160                this.initialOfferedTraffic_d = initialNetPlan.getVectorDemandOfferedTraffic(trafficLayer);
161                this.currentTheoreticalOfferedTraffic_d = initialNetPlan.getVectorDemandOfferedTraffic(trafficLayer);
162                this.isCac = (_trafficType.getString ().equalsIgnoreCase("connection-based-longrun") || _trafficType.getString ().equalsIgnoreCase("connection-based-incremental"));
163                /* Initialize CAC if applicable */
164                if (isCac)
165                {
166                        this.cac_auxIATDeterministic = cac_arrivalsPattern.getString ().equalsIgnoreCase("deterministic");
167                        this.cac_auxIATExponential = cac_arrivalsPattern.getString ().equalsIgnoreCase("random-exponential-arrivals-deterministic-duration") || cac_arrivalsPattern.getString ().equalsIgnoreCase("random-exponential-arrivals-and-duration");
168                        this.cac_auxDurationDeterministic = cac_arrivalsPattern.getString ().equalsIgnoreCase("deterministic") || cac_arrivalsPattern.getString ().equalsIgnoreCase("random-exponential-arrivals-deterministic-duration");
169                        this.cac_auxDurationExponential = cac_arrivalsPattern.getString ().equalsIgnoreCase("random-exponential-arrivals-and-duration");
170                        this.cac_auxIncremental = _trafficType.getString ().equalsIgnoreCase("connection-based-incremental");
171                        this.cac_avHoldingTimeHours_d = DoubleFactory1D.dense.make (D , 0); 
172                        this.cac_avConnectionSize_d = DoubleFactory1D.dense.make (D , 0);
173                        this.cacIncremental_potentiallyBlockedRouteRequests = cac_auxIncremental? new HashSet<Pair<WDMUtils.LightpathAdd,Double>> () : null;
174                        for (Demand originalDemand : initialNetPlan.getDemands(trafficLayer))
175                        {
176                                final int d = originalDemand.getIndex();
177                                final double averageConnectionSize = DoubleUtils.scalarProduct(lineRatesGbps_t , probabilitiesLineRates_t);
178                                final double holdingTime = (originalDemand.getAttribute("holdingTime") != null)? Double.parseDouble(originalDemand.getAttribute("holdingTime")) : cac_avHoldingTimeHours.getDouble();
179                                final double avIATHours = averageConnectionSize * holdingTime / currentTheoreticalOfferedTraffic_d.get(d);
180                                final double nextInterArrivalTime = cac_auxIATDeterministic? avIATHours : cac_auxIATExponential? Exponential.staticNextDouble(1/avIATHours) : -1;
181                                cac_avHoldingTimeHours_d.set (d,holdingTime);
182                                cac_avConnectionSize_d.set (d,averageConnectionSize);
183                                scheduleEvent(new SimEvent(nextInterArrivalTime, SimEvent.DestinationModule.EVENT_GENERATOR , -1 , new GenerateConnectionRequest(originalDemand)));
184                        }
185                }
186                
187                /* Initialize fast changing traffic */
188                if (_tfFast_fluctuationType.getString ().equalsIgnoreCase("random-truncated-gaussian"))
189                {
190                        this.tfFast_auxRandomGaussian = true;
191                        for (Demand originalDemand : initialNetPlan.getDemands(trafficLayer))
192                                scheduleEvent(new SimEvent(0, SimEvent.DestinationModule.EVENT_GENERATOR , -1 , new GenerateDemandOfferedTrafficFastFluctuation(originalDemand)));
193                }
194
195                /* Initialize slow changing traffic */
196                this.slowChangingOfferedTraffic_d = initialNetPlan.getVectorDemandOfferedTraffic(trafficLayer); // the offered traffic is the sum of the two
197                if (_tfSlow_fluctuationType.getString ().equalsIgnoreCase("time-zone-based"))
198                {
199                        this.tfSlow_auxTimeZoneBased = true;
200                        this.tfSlow_calendar = Calendar.getInstance(); 
201                        try { this.tfSlow_calendar.setTime(new SimpleDateFormat(DATE_FORMAT).parse(tfSlow_startDate.getString())); } catch (Exception e) { e.printStackTrace(); throw new Net2PlanException ("Error parsing the date"); }
202                        this.tfSlow_simTimeOfLastCalendarUpdate = 0;
203                        tfSlow_timeZones_n = DoubleFactory1D.dense.make (N , tfSlow_defaultTimezone.getInt());
204                        for(Node node : initialNetPlan.getNodes ())
205                        {
206                                if (node.getAttribute("timezone") == null) continue;
207                                double timezone = Double.parseDouble(node.getAttribute("timezone"));
208                                if (timezone < -12 || timezone > 12) throw new Net2PlanException(String.format("Timezone for node %d must be in range [-12, 12]", node.getIndex ()));
209                                tfSlow_timeZones_n.set(node.getIndex (), timezone);
210                        }
211                        for (Demand demand : initialNetPlan.getDemands(trafficLayer))
212                                scheduleEvent(new SimEvent(0.0, SimEvent.DestinationModule.EVENT_GENERATOR , -1 , new GenerateDemandOfferedTrafficSlowFluctuation(demand)));
213                }
214
215                /* Initialize slow changing traffic */
216                if (!_fail_failureModel.getString ().equalsIgnoreCase("none"))
217                {
218                        this.fail_currentlyFailedSRGs = new HashSet<SharedRiskGroup> ();
219                        if (!fail_statisticalPattern.getString ().equalsIgnoreCase("exponential-iid")) throw new Net2PlanException ("Unknown failure statisitical pattern");
220                        switch (_fail_failureModel.getString ())
221                        {
222                                case "SRGfromNetPlan":
223                                        break;
224                                case "perNode":
225                                        SRGUtils.configureSRGs(initialNetPlan, fail_defaultMTTFInHours.getDouble(), fail_defaultMTTRInHours.getDouble(), SRGUtils.SharedRiskModel.PER_NODE, true);
226                                        break;
227                                case "perLink":
228                                        SRGUtils.configureSRGs(initialNetPlan, fail_defaultMTTFInHours.getDouble(), fail_defaultMTTRInHours.getDouble(), SRGUtils.SharedRiskModel.PER_LINK, true);
229                                        break;
230                                case "perDirectionalLinkBundle":
231                                        SRGUtils.configureSRGs(initialNetPlan, fail_defaultMTTFInHours.getDouble(), fail_defaultMTTRInHours.getDouble(), SRGUtils.SharedRiskModel.PER_DIRECTIONAL_LINK_BUNDLE, true);
232                                        break;
233                                case "perBidirectionalLinkBundle":
234                                        SRGUtils.configureSRGs(initialNetPlan, fail_defaultMTTFInHours.getDouble(), fail_defaultMTTRInHours.getDouble(), SRGUtils.SharedRiskModel.PER_BIDIRECTIONAL_LINK_BUNDLE, true);
235                                        break;
236                                default:
237                                        throw new Net2PlanException("Failure model not valid. Please, check algorithm parameters description");
238                        }
239                        if (initialNetPlan.getNumberOfSRGs() == 0) throw new Net2PlanException("No SRGs were defined");
240                        for (SharedRiskGroup srg : initialNetPlan.getSRGs())
241                        {
242                                final double nextEvent = Exponential.staticNextDouble(1 / srg.getMeanTimeToFailInHours());
243                                scheduleEvent(new SimEvent(nextEvent , SimEvent.DestinationModule.EVENT_GENERATOR , -1 , new GenerateFailureSRG(srg)));
244                        }
245                }
246
247
248        }

250        @Override
251        public void processEvent(NetPlan currentNetPlan, SimEvent event)
252        {
253                final double simTime = event.getEventTime();
254                Object eventObject = event.getEventObject();
255
256                /* if a connection could not be setup, end simulation in the incremental simulation mode */
257                if (this.cac_auxIncremental)
258                {
259                        for (Pair<WDMUtils.LightpathAdd,Double> ev : new LinkedList<Pair<WDMUtils.LightpathAdd,Double>> (this.cacIncremental_potentiallyBlockedRouteRequests))
260                        {
261                                if (ev.getFirst().lpAddedToFillByProcessor != null) 
262                                        cacIncremental_potentiallyBlockedRouteRequests.remove(ev);
263                                else if (ev.getSecond() < simTime) endSimulation(); // not assigned route, and it is in the past => end simulation in the incremental mode
264                        }
265                }
266                
267                if (eventObject instanceof GenerateConnectionRequest)
268                {
269                        final GenerateConnectionRequest connectionRequest = (GenerateConnectionRequest) eventObject;
270                        final Demand demand = connectionRequest.demand;
271                        final int d = demand.getIndex ();
272                        final double h_d = currentTheoreticalOfferedTraffic_d.get(d); // same traffic units as connection size
273                        final double avHoldingTimeHours = cac_avHoldingTimeHours_d.get(d);
274                        final double connectionSize = cac_avConnectionSize_d.get (d);
275                        final double avIATHours = connectionSize * avHoldingTimeHours / h_d;
276                        final double nextHoldingTime = cac_auxDurationDeterministic? avHoldingTimeHours : cac_auxDurationExponential? Exponential.staticNextDouble(1/avHoldingTimeHours) : -1;
277                        final double nextInterArrivalTime = cac_auxIATDeterministic? avIATHours : cac_auxIATExponential? Exponential.staticNextDouble(1/avIATHours) : -1;
278                        final double lineRateThisLpGbps = randomPick (lineRatesGbps_t , probabilitiesLineRates_t);
279                        
280                        /* Events to the processor. RouteAdd, and if not incremental mode, route remove */
281                        WDMUtils.LightpathAdd routeInfo_add = new WDMUtils.LightpathAdd(demand , lineRateThisLpGbps);
282                        scheduleEvent(new SimEvent (simTime, SimEvent.DestinationModule.EVENT_PROCESSOR , -1 , routeInfo_add));
283                        if (cac_auxIncremental)
284                                this.cacIncremental_potentiallyBlockedRouteRequests.add (Pair.of(routeInfo_add,simTime)); // to check later if it was blocked
285                        else
286                                scheduleEvent(new SimEvent(simTime + nextHoldingTime, SimEvent.DestinationModule.EVENT_GENERATOR , -1 , new GenerateConnectionRelease(routeInfo_add)));
287                        
288                        /* Event for me: new connection */
289                        scheduleEvent(new SimEvent(simTime + nextInterArrivalTime, SimEvent.DestinationModule.EVENT_GENERATOR , -1 , new GenerateConnectionRequest(demand)));
290                }
291                if (eventObject instanceof GenerateConnectionRelease)
292                {
293                        final GenerateConnectionRelease releaseEvent = (GenerateConnectionRelease) eventObject;
294//                      SimEvent.DemandModify demandOfferedTrafficUpdate = new SimEvent.DemandModify(releaseEvent.routeAddEvent.demand , -releaseEvent.routeAddEvent.carriedTraffic , true);
295//                      scheduleEvent(new SimEvent (simTime, SimEvent.DestinationModule.EVENT_PROCESSOR , -1 , demandOfferedTrafficUpdate));
296                        if (releaseEvent.routeAddEvent.lpAddedToFillByProcessor != null)
297                        {
298                                WDMUtils.LightpathRemove routeInfo_remove = new WDMUtils.LightpathRemove(releaseEvent.routeAddEvent.lpAddedToFillByProcessor);
299                                scheduleEvent(new SimEvent (simTime , SimEvent.DestinationModule.EVENT_PROCESSOR , -1 , routeInfo_remove));
300                        }
301                }
302                else if (eventObject instanceof GenerateDemandOfferedTrafficFastFluctuation)
303                {
304                        final GenerateDemandOfferedTrafficFastFluctuation trafficFluctuation = (GenerateDemandOfferedTrafficFastFluctuation) eventObject;
305                        final Demand demand = trafficFluctuation.demand;
306                        final int d = demand.getIndex ();
307                        final double slowChangingTrafficPart = slowChangingOfferedTraffic_d.get(d);
308                        if (tfFast_auxRandomGaussian)
309                        {
310                                double newFastTrafficVariation = rng.nextGaussian() * tfFast_fluctuationCoefficientOfVariation.getDouble() * slowChangingTrafficPart;
311                                newFastTrafficVariation = Math.max (newFastTrafficVariation , slowChangingTrafficPart * (1 - tfFast_maximumFluctuationRelativeFactor.getDouble()));
312                                newFastTrafficVariation = Math.min (newFastTrafficVariation , slowChangingTrafficPart * (1 + tfFast_maximumFluctuationRelativeFactor.getDouble()));
313                                currentTheoreticalOfferedTraffic_d.set (d , slowChangingTrafficPart + newFastTrafficVariation);
314                                if (!isCac) // inform the processor with a demand modified only if it is NOT cac. In CAC the sent events are the routes only, and the algorithms update the offered traffic according to it
315                                {
316                                        SimEvent.DemandModify modifyEvent = new SimEvent.DemandModify(demand , Math.max (0 , slowChangingTrafficPart + newFastTrafficVariation) , false);
317                                        scheduleEvent(new SimEvent (simTime, SimEvent.DestinationModule.EVENT_PROCESSOR , -1 , modifyEvent));
318                                }
319                        }
320                        else if (_tfFast_fluctuationType.getString ().equalsIgnoreCase("none"))
321                        {
322                                throw new RuntimeException ("Bad");
323                        }
324                        else throw new Net2PlanException ("Unknow fast traffic fluctuation type: " + _tfFast_fluctuationType.getString ());
325                        /* Send event to me for the next fast change */
326                        scheduleEvent(new SimEvent(simTime + tfFast_timeBetweenDemandFluctuationsHours.getDouble() , SimEvent.DestinationModule.EVENT_GENERATOR , -1 , new GenerateDemandOfferedTrafficFastFluctuation(demand)));
327                }
328                else if (eventObject instanceof GenerateDemandOfferedTrafficSlowFluctuation)
329                {
330                        final GenerateDemandOfferedTrafficSlowFluctuation trafficFluctuation = (GenerateDemandOfferedTrafficSlowFluctuation) eventObject;
331                        final Demand demand = trafficFluctuation.demand;
332                        final int d = demand.getIndex ();
333                        final double currentSlowHd = slowChangingOfferedTraffic_d.get(d);
334                        final double currentHd = currentTheoreticalOfferedTraffic_d.get(d);
335                        if (tfSlow_auxTimeZoneBased)
336                        {
337                                /* Send event to processor with the demand change */
338                                tfSlow_calendar.add(Calendar.MILLISECOND, (int) ((simTime - tfSlow_simTimeOfLastCalendarUpdate) * 1000));
339                                final int hours = tfSlow_calendar.get(Calendar.HOUR_OF_DAY);
340                                final int minutes = tfSlow_calendar.get(Calendar.MINUTE);
341                                final int seconds = tfSlow_calendar.get(Calendar.SECOND);
342                                final int weekday = tfSlow_calendar.get(Calendar.DAY_OF_WEEK);
343                                final double UTC = hours + (double) minutes / 60 + (double) seconds / 3600;
344                                final double peakTrafficFactor = weekday == Calendar.SATURDAY || weekday == Calendar.SUNDAY ? 0.5 : 1;
345                                final double activityOriginNode = TrafficMatrixGenerationModels.activityFactor(UTC, tfSlow_timeZones_n.get(demand.getIngressNode().getIndex ()), 0.3, peakTrafficFactor);
346                                final double activityDestinationNode = TrafficMatrixGenerationModels.activityFactor(UTC, tfSlow_timeZones_n.get(demand.getEgressNode().getIndex ()), 0.3, peakTrafficFactor);
347                                final double activityFactorNodePair = Math.max (0 , (activityOriginNode + activityDestinationNode) / 2);
348                                final double newSlowFluctuationTraffic = initialOfferedTraffic_d.get(d) * activityFactorNodePair;
349                                final double currentFastFluctuationTraffic = currentHd - currentSlowHd;
350                                currentTheoreticalOfferedTraffic_d.set (d , newSlowFluctuationTraffic + currentFastFluctuationTraffic);
351                                if (!isCac) // inform the processor with a demand modified only if it is NOT cac. In CAC the sent events are the routes only, and the algorithms update the offered traffic according to it
352                                {
353                                        SimEvent.DemandModify modifyEvent = new SimEvent.DemandModify(demand , Math.max (0 , newSlowFluctuationTraffic + currentFastFluctuationTraffic), false);
354                                        scheduleEvent(new SimEvent (simTime, SimEvent.DestinationModule.EVENT_PROCESSOR , -1 , modifyEvent));
355                                }
356                                tfSlow_simTimeOfLastCalendarUpdate = simTime;
357                        }
358                        else throw new Net2PlanException ("Unknow fast traffic fluctuation type: " + _tfFast_fluctuationType.getString ());
359                        /* Send event to me for the next fast change */
360                        scheduleEvent(new SimEvent(simTime + tfSlow_timeBetweenDemandFluctuationsHours.getDouble() , SimEvent.DestinationModule.EVENT_GENERATOR , -1 , new GenerateDemandOfferedTrafficSlowFluctuation(demand)));
361                }
362                else if (eventObject instanceof GenerateFailureSRG)
363                {
364                        final GenerateFailureSRG srgEvent = (GenerateFailureSRG) eventObject;
365                        final SharedRiskGroup srg = srgEvent.srg;
366                        
367                        /* Send event of appropriate failures to the processor (only links and nodes changing its state) */
368                        Set<Node> nodesUpToDown = new HashSet<Node> (currentNetPlan.getNodesUp()); nodesUpToDown.retainAll(srg.getNodes());
369                        Set<Link> linksUpToDown = new HashSet<Link> (currentNetPlan.getLinksUpAllLayers()); linksUpToDown.retainAll(srg.getLinks());
370                        if (!nodesUpToDown.isEmpty() || !linksUpToDown.isEmpty())
371                        {
372                                SimEvent.NodesAndLinksChangeFailureState failEvent = new SimEvent.NodesAndLinksChangeFailureState (null , nodesUpToDown , null , linksUpToDown);
373                                scheduleEvent(new SimEvent(simTime , SimEvent.DestinationModule.EVENT_PROCESSOR , -1 , failEvent));
374                        }
375                        /* Send repair event to myself */
376                        scheduleEvent(new SimEvent(simTime + Exponential.staticNextDouble(1 / srg.getMeanTimeToRepairInHours()) , SimEvent.DestinationModule.EVENT_GENERATOR , -1 , new GenerateRepairSRG(srg)));                       
377                        
378                        fail_currentlyFailedSRGs.add (srg);
379                }
380                else if (eventObject instanceof GenerateRepairSRG)
381                {
382                        final GenerateRepairSRG srgEvent = (GenerateRepairSRG) eventObject;
383                        final SharedRiskGroup srg = srgEvent.srg;
384                        
385                        /* Send event of appropriate repairs to the processor (only links and nodes changing its state) */
386                        fail_currentlyFailedSRGs.remove (srg);
387                        Set<Node> nodesDownAfterRepair = new HashSet<Node> (); 
388                        Set<Link> linksDownAfterRepair = new HashSet<Link> (); 
389                        for (SharedRiskGroup srgStillFailed : fail_currentlyFailedSRGs)
390                        {
391                                nodesDownAfterRepair.addAll (srgStillFailed.getNodes());
392                                linksDownAfterRepair.addAll (srgStillFailed.getLinks());
393                        }
394                        Set<Node> nodesDownToUp = new HashSet<Node> (currentNetPlan.getNodesDown()); nodesDownToUp.removeAll (nodesDownAfterRepair);
395                        Set<Link> linksDownToUp = new HashSet<Link> (currentNetPlan.getLinksDownAllLayers()); linksDownToUp.removeAll (linksDownAfterRepair);
396                        
397                        if (!nodesDownToUp.isEmpty() || !linksDownToUp.isEmpty())
398                        {
399                                SimEvent.NodesAndLinksChangeFailureState repairEvent = new SimEvent.NodesAndLinksChangeFailureState (nodesDownToUp , null , linksDownToUp , null);
400                                scheduleEvent(new SimEvent(simTime , SimEvent.DestinationModule.EVENT_PROCESSOR , -1 , repairEvent));
401                        }
402                        /* Send repair event to myself */
403                        scheduleEvent(new SimEvent(simTime + Exponential.staticNextDouble(1 / srg.getMeanTimeToFailInHours()) , SimEvent.DestinationModule.EVENT_GENERATOR , -1 , new GenerateFailureSRG(srg)));                        
404                }
405        }

407        private double randomPick (double [] vals , double [] accumProbs)
408        {
409                final double x = rng.nextDouble();
410                for (int cont = 0 ; cont < vals.length-1 ; cont ++)
411                        if (accumProbs [cont] < x) return vals [cont];
412                return vals [vals.length - 1];
413        }
