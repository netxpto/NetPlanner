048/** Implements the reactions of a WDM network carrying lightpaths in a fixed or flexi grid of wavelengths. 
049 * 
050 * <p>The design follows the assumptions described in {@link com.net2plan.libraries.WDMUtils WDMUtils} Net2Plan library</p>
051 * <p>This algorithm implements the reactions of a WDM network carrying lightpaths, to the following events:</p> 
052 * <ul>
053 * <li>{@link com.net2plan.libraries.WDMUtils.LightpathAdd WDMUtils.LightpathAdd}: Adds the corresponding lightpath to the network 
054 * (the Route object and potentially a ProtectionSegment object if the lightpath is asked to be 1+1 protected), 
055 * if enough resources exist for it. If the event includes a Demand object, the lightpath is associated to it. 
056 * If not, a new demand is created. The object establishes the line rate of the lightpath to establish. If such line rate is 
057 * not present in any transponder type defined, an exception is raised (on valid transponders exist for such a lightpath).
058 * If the event includes the RSA (also of the backup, if 1+1 protection is selected), the algorithm 
059 * tries this RSA. If not, the user-selected strategy for the routing and spectrum assignment of the lightpath is applied.
060 * Such strategy is tried first with the first transponder type (in order) with the appropriate line rate. If a valid RSA is not found 
061 * it is repeated with the next. The lightpath request is blocked if under no transponder type of the appropriate line rate, a RSA 
062 * is found (note that different transponders can have e.g. different optical reaches or number of occupied slots).</li>
063 * <li>{@link com.net2plan.libraries.WDMUtils.LightpathModify WDMUtils.LightpathModify}: Modifies the carried traffic and/or the RSA of a lightpath.</li>
064 * <li>{@link com.net2plan.libraries.WDMUtils.LightpathRemove WDMUtils.LightpathRemove}: Removes the corresponding lightpath (including the Demand and Route objects, and potentially the 1+1 segment if any), releasing the resources.</li>
065 * <li>{@link com.net2plan.interfaces.simulation.SimEvent.NodesAndLinksChangeFailureState SimEvent.NodesAndLinksChangeFailureState}: Fails/repairs the indicated nodes and/or links, and reacts to such failures 
066 * (the particular form depends on the network recovery options selected).</li>
067 * <li>{@link com.net2plan.interfaces.simulation.SimEvent.DemandModify SimEvent.DemandModify}: Modifies the offered traffic of a demand.</li>
068 * <li>{@link com.net2plan.interfaces.simulation.SimEvent.DemandRemove SimEvent.DemandRemove}: Removes a demand, and all its associated lightpaths if any, releasing the resources.</li>
069 * </ul>
070 * 
071 * This module can be used in conjunction with the {@code Online_evGen_wdm} generator for creating the events to which 
072 * this module reacts. 
073 * 
074 * See in {@link com.net2plan.libraries.WDMUtils} Javadoc the WDM technology conventions used in Net2Plan built-in algorithms and libraries to represent WDM networks. 
075 * @net2plan.keywords WDM, Network recovery: protection, Network recovery: restoration
076 * @net2plan.inputParameters 
077 */
078public class Online_evProc_wdm extends IEventProcessor

127        @Override
128        public void initialize(NetPlan initialNetPlan, Map<String, String> algorithmParameters, Map<String, String> simulationParameters, Map<String, String> net2planParameters)
129        {
130                /* Initialize all InputParameter objects defined in this object (this uses Java reflection) */
131                InputParameter.initializeAllInputParameterFieldsOfObject(this, algorithmParameters);
132                
133                this.wdmLayer = wdmLayerIndex.getInt () == -1? initialNetPlan.getNetworkLayerDefault() : initialNetPlan.getNetworkLayer(wdmLayerIndex.getInt ());
134                this.wdmRouteOriginalRwa = new HashMap<Route,Pair<WDMUtils.RSA,WDMUtils.RSA>> ();
135                this.isRestorationRecovery = wdmRecoveryType.getString ().equalsIgnoreCase("restoration");
136                this.isProtectionRecovery = wdmRecoveryType.getString ().equalsIgnoreCase("protection");
137                this.isAlternateRouting = wdmRwaType.getString().equalsIgnoreCase("alternate-routing");
138                this.isLeastCongestedRouting = wdmRwaType.getString().equalsIgnoreCase("least-congested-routing");
139                this.isSrgDisjointAwareLpRouting =      wdmRwaType.getString().equalsIgnoreCase("srg-disjointness-aware-route-first-fit");;
140                this.isLoadSharing = wdmRwaType.getString().equalsIgnoreCase("load-sharing");
141                this.newRoutesHave11Protection = !wdmProtectionTypeToNewRoutes.getString ().equalsIgnoreCase("none");
142                this.rng = new Random(wdmRandomSeed.getLong () == -1? (long) RandomUtils.random(0, Long.MAX_VALUE - 1) : wdmRandomSeed.getLong ());
143                if (!isProtectionRecovery && newRoutesHave11Protection) throw new Net2PlanException ("In the input parameter you ask to assign protection paths to new connections, while the recovery type chosen does not use them");
144                
145                
146                if (wdmRemovePreviousLightpaths.getBoolean())
147                {
148                        initialNetPlan.removeAllRoutes(wdmLayer);
149                        initialNetPlan.removeAllProtectionSegments(wdmLayer);
150                        initialNetPlan.removeAllMulticastTrees(wdmLayer);
151                }
152                initialNetPlan.setRoutingType(RoutingType.SOURCE_ROUTING , wdmLayer);
153                if (wdmNumFrequencySlotsPerFiber.getInt() > 0) WDMUtils.setFibersNumFrequencySlots(initialNetPlan , wdmNumFrequencySlotsPerFiber.getInt() , wdmLayer);
154                
155                /* Get the transponders information */
156                this.tpInfo = new TransponderTypesInfo(wdmTransponderTypesInfo.getString());
157                this.transponderTypeOfNewLps = new HashMap<Route,Integer> ();
158
159                /* Compute the candidate path list */
160                this.E_wdm = initialNetPlan.getNumberOfLinks(wdmLayer);
161                this.cplWdm = initialNetPlan.copy(); 
162                for (NetworkLayer cplLayer : new HashSet<NetworkLayer> (cplWdm.getNetworkLayers())) if (cplLayer.getId () != wdmLayer.getId ()) cplWdm.removeNetworkLayer(cplLayer);
163                cplWdm.removeAllDemands();
164                for (Node n1 : cplWdm.getNodes ()) for (Node n2 : cplWdm.getNodes ()) if (n1 != n2) cplWdm.addDemand(n1 , n2 , 0 , null);
165                final Map<Demand,List<List<Link>>> cplPaths = cplWdm.computeUnicastCandidatePathList(initialNetPlan.getVectorLinkLengthInKm(wdmLayer).toArray() , "K", Integer.toString(wdmK.getInt ()), "maxLengthInKm", Double.toString(tpInfo.getMaxOpticalReachKm()) , "maxNumHops", Integer.toString(wdmMaxLightpathNumHops.getInt () > 0? wdmMaxLightpathNumHops.getInt () : Integer.MAX_VALUE));
166                this.protectionTypeCode = wdmProtectionTypeToNewRoutes.getString ().equals("1+1-srg-disjoint") ? 0 : wdmProtectionTypeToNewRoutes.getString ().equals("1+1-node-disjoint")? 1 : 2;
167                if (newRoutesHave11Protection)
168                {
169                        final Map<Demand,List<Pair<List<Link>,List<Link>>>> pathPairs = NetPlan.computeUnicastCandidate11PathList(cplPaths, protectionTypeCode);
170                        this.cplWdm.addRoutesAndProtectionSegmentFromCandidate11PathList(pathPairs);
171                        for (Route r : cplWdm.getRoutes()) 
172                        { 
173                                checkDisjointness (r.getSeqLinksRealPath() , r.getPotentialBackupProtectionSegments().iterator().next().getSeqLinks() , protectionTypeCode); 
174                        }
175                }
176                else
177                {
178                        this.cplWdm.addRoutesFromCandidatePathList(cplPaths);
179                }
180                this.cplA_rs = cplWdm.getMatrixRoute2SRGAffecting();
181                this.cplA_pss = cplWdm.getMatrixProtectionSegment2SRGAffecting();
182
183                this.wavelengthFiberOccupancy = WDMUtils.getNetworkSlotAndRegeneratorOcupancy(initialNetPlan, true , wdmLayer).getFirst();
184                if (DEBUG) checkClashing (initialNetPlan); 
185                initialNetPlan.setLinkCapacityUnitsName("Frequency slots" , wdmLayer);
186
187                for (Route r : initialNetPlan.getRoutes(wdmLayer))
188                {
189                        if (!r.getSeqLinksAndProtectionSegments().equals (r.getSeqLinksRealPath())) throw new Net2PlanException ("Some of the initial routes are traversing protection segments");
190
191                        WDMUtils.RSA thisRouteRWA = new WDMUtils.RSA (r , false);
192                        if (thisRouteRWA.hasFrequencySlotConversions()) throw new Net2PlanException ("Initial lightpaths must have route continuity");
193                        WDMUtils.RSA segmentRWA = null;
194                        if (r.getPotentialBackupProtectionSegments().size () > 1) throw new Net2PlanException ("The number of protection segments of a route cannot be higher than one");
195                        if ((r.getPotentialBackupProtectionSegments().size () == 1) && (!isProtectionRecovery)) throw new Net2PlanException ("Initial routes have protection segments, which would be unused since the recovery type is not protection");
196                        if (r.getPotentialBackupProtectionSegments().size () == 1)
197                        {
198                                final ProtectionSegment backupPath = r.getPotentialBackupProtectionSegments().iterator().next();
199                                segmentRWA = new WDMUtils.RSA (backupPath);
200                                if (segmentRWA.hasFrequencySlotConversions()) throw new Net2PlanException ("Initial lightpaths must have route continuity");
201                        }                       
202                        wdmRouteOriginalRwa.put (r , Pair.of (thisRouteRWA , segmentRWA));
203                }
204                this.finishTransitory(0);
205                if (DEBUG) checkClashing (initialNetPlan); 
206        }

208        @Override
209        public void processEvent(NetPlan currentNetPlan, SimEvent event)
210        {
211                try {
212                        if (event.getEventObject () instanceof WDMUtils.LightpathAdd)
213                        {
214                                WDMUtils.LightpathAdd addLpEvent = (WDMUtils.LightpathAdd) event.getEventObject ();
215                                if (addLpEvent.layer != this.wdmLayer) throw new Net2PlanException ("Lightpaths cannot be added to layer " + addLpEvent.demand.getLayer() + ", and just to the WDM layer (" + wdmLayer + ")");
216                                final Node ingressNode = addLpEvent.ingressNode;
217                                final Node egressNode = addLpEvent.egressNode;
218                                final Demand cplDemand = this.cplWdm.getNodePairDemands(cplWdm.getNodeFromId(ingressNode.getId()) , cplWdm.getNodeFromId(egressNode.getId()) , false).iterator().next();
219                                final double lineRateThisLp_Gbps = addLpEvent.lineRateGbps;
220                                if (!tpInfo.isValidLineRateForAtLeastOneType (lineRateThisLp_Gbps)) throw new Net2PlanException ("Requested to set up a lightpath of a line rate (" + lineRateThisLp_Gbps + ") for which I don't have transpoders");
221
222//                              System.out.println ("EVENT LightpathAddAsPrimary: " + addLpEvent + ", demand: " + addLpEvent.demand);
223                                /* update the offered traffic of the demand */
224                                this.stat_numOfferedConnections ++;
225                                this.stat_trafficOfferedConnections += lineRateThisLp_Gbps;
226                                
227                                /* Computes one or two paths over the links (the second path would be a segment). You cannot use the already existing segments in these paths */
228                                if (newRoutesHave11Protection)
229                                {
230                                        /* The RWA may be computed by me, or mandated by the event */
231                                        Pair<WDMUtils.RSA,WDMUtils.RSA> rwa = null;
232                                        int transponderTypeUsed = -1;
233                                        if ((addLpEvent.primaryRSA != null) && (addLpEvent.backupRSA != null))
234                                        {
235                                                if (!WDMUtils.isAllocatableRSASet(wavelengthFiberOccupancy , addLpEvent.primaryRSA , addLpEvent.backupRSA))
236                                                        throw new Net2PlanException ("A request was received to add a conflicting lightpath 1+1 pair");
237                                                rwa = Pair.of (addLpEvent.primaryRSA , addLpEvent.backupRSA);
238                                        }
239                                        else
240                                        {
241                                                /* Check among the transponder types (in order), the one in the same rate that fits (optical reach in range). */
242                                                for (int t = 0; t < tpInfo.getNumTypes() ; t ++)
243                                                {
244                                                        if (lineRateThisLp_Gbps != tpInfo.getLineRateGbps(t)) continue;
245                                                        rwa = computeValid11PathPairNewRoute(cplDemand , currentNetPlan , tpInfo.getNumSlots(t) , tpInfo.getOpticalReachKm(t) , tpInfo.isOpticalRegenerationPossible(t));
246                                                        if (rwa != null) { transponderTypeUsed = t; break; }
247                                                }
248                                        }
249
250//                              System.out.println ("------ ADD LP: rwa: " + rwa);
251                                        if (rwa != null)
252                                        {
253                                                if (!WDMUtils.isAllocatableRSASet(wavelengthFiberOccupancy , rwa.getFirst() , rwa.getSecond())) throw new RuntimeException ("Bad");
254                                                for (Link e : rwa.getFirst ().seqLinks) if (e.getNetPlan() != currentNetPlan) throw new RuntimeException ("Bad");
255                                                for (Link e : rwa.getSecond ().seqLinks) if (e.getNetPlan() != currentNetPlan) throw new RuntimeException ("Bad");
256                                                
257                                                final Demand wdmLayerDemand = addLpEvent.demand == null? currentNetPlan.addDemand(addLpEvent.ingressNode, addLpEvent.egressNode, lineRateThisLp_Gbps , null, wdmLayer) : addLpEvent.demand;
258                                                final Route wdmLayerRoute = WDMUtils.addLightpath(wdmLayerDemand, rwa.getFirst(), lineRateThisLp_Gbps);
259                                                WDMUtils.allocateResources(rwa.getFirst() , wavelengthFiberOccupancy , null);
260                                                if (rwa.getFirst().seqLinks.equals(rwa.getSecond().seqLinks)) throw new RuntimeException ("Both 1+1 same route");
261                                                final ProtectionSegment wdmLayerSegment = WDMUtils.addLightpathAsProtectionSegment(rwa.getSecond());
262                                                WDMUtils.allocateResources(rwa.getSecond() , wavelengthFiberOccupancy , null);
263                                                checkDisjointness(wdmLayerRoute.getSeqLinksRealPath() , wdmLayerSegment.getSeqLinks() , protectionTypeCode);
264                                                wdmLayerRoute.addProtectionSegment(wdmLayerSegment);
265                                                this.wdmRouteOriginalRwa.put (wdmLayerRoute , rwa);
266                                                this.transponderTypeOfNewLps.put(wdmLayerRoute , transponderTypeUsed);
267                                                this.stat_numCarriedConnections ++;
268                                                this.stat_trafficCarriedConnections += lineRateThisLp_Gbps;
269                                                addLpEvent.lpAddedToFillByProcessor = wdmLayerRoute;
270                                                if (DEBUG) checkClashing (currentNetPlan); 
271                                        }
272                                        if (DEBUG) checkClashing (currentNetPlan); 
273                                }
274                                else
275                                {
276                                        /* The RWA may be computed by me, or mandated by the event */
277                                        WDMUtils.RSA rwa = null;
278                                        int transponderTypeUsed = -1;
279                                        if (addLpEvent.primaryRSA != null)
280                                        {
281                                                if (!WDMUtils.isAllocatableRSASet(wavelengthFiberOccupancy , addLpEvent.primaryRSA))
282                                                        throw new Net2PlanException ("A request was received to add a conflicting lightpath");
283                                                rwa = addLpEvent.primaryRSA; 
284                                        }
285                                        else
286                                        {
287                                                /* Check among the transponder types (in order), the one in the same rate that fits (optical reach in range). */
288                                                for (int t = 0; t < tpInfo.getNumTypes() ; t ++)
289                                                {
290                                                        if (lineRateThisLp_Gbps != tpInfo.getLineRateGbps(t)) continue;
291                                                        rwa = computeValidPathNewRoute(cplDemand , currentNetPlan , tpInfo.getNumSlots(t) , tpInfo.getOpticalReachKm(t) , tpInfo.isOpticalRegenerationPossible(t));
292                                                        if (rwa != null) { transponderTypeUsed = t; break; }
293                                                }
294                                        }
295//                              System.out.println ("------ ADD LP: rwa: " + rwa);
296                                        if (rwa != null)
297                                        {
298                                                final Demand wdmLayerDemand = addLpEvent.demand == null? currentNetPlan.addDemand(addLpEvent.ingressNode, addLpEvent.egressNode, lineRateThisLp_Gbps , null, wdmLayer) : addLpEvent.demand;
299                                                final Route wdmLayerRoute = WDMUtils.addLightpath(wdmLayerDemand, rwa , lineRateThisLp_Gbps);
300                                                WDMUtils.allocateResources(rwa , wavelengthFiberOccupancy , null);
301                                                this.wdmRouteOriginalRwa.put (wdmLayerRoute , Pair.of(rwa,(WDMUtils.RSA)null));
302                                                this.transponderTypeOfNewLps.put(wdmLayerRoute , transponderTypeUsed);
303                                                this.stat_numCarriedConnections ++;
304                                                this.stat_trafficCarriedConnections += lineRateThisLp_Gbps;
305                                                addLpEvent.lpAddedToFillByProcessor = wdmLayerRoute;
306//                                      System.out.println ("Added lp: " + wdmLayerRoute);
307                                        }
308                                }
309                                if (DEBUG) checkClashing (currentNetPlan); 
310                        }
311                        else if (event.getEventObject () instanceof WDMUtils.LightpathRemove)
312                        {
313                                WDMUtils.LightpathRemove lpEvent = (WDMUtils.LightpathRemove) event.getEventObject ();
314                                final Route lpToRemove = lpEvent.lp;
315                                final Pair<WDMUtils.RSA,WDMUtils.RSA> originalRwa = wdmRouteOriginalRwa.get(lpToRemove);
316                                if (originalRwa == null) return; // the ligtpath was already removed, because it was first accepted, but then blocked because of failures 
317                                if (isRestorationRecovery) // restoration => the current route is the one to release
318                                        removeRoute_restoration(lpToRemove);
319                                else
320                                        removeRoute_protection(lpToRemove , originalRwa);
321                                this.transponderTypeOfNewLps.remove(lpToRemove);
322                                if (DEBUG) checkClashing (currentNetPlan); 
323                        } else if (event.getEventObject () instanceof SimEvent.NodesAndLinksChangeFailureState)
324                        {
325                                SimEvent.NodesAndLinksChangeFailureState ev = (SimEvent.NodesAndLinksChangeFailureState) event.getEventObject ();
326
327//                              System.out.println ("----------- Event NodesAndLinksChangeFailureState: links up" + ev.linksToUp + ", links down: " + ev.linksToDown);
328
329                                if (DEBUG) checkClashing (currentNetPlan); 
330
331                                /* This automatically sets as up the routes affected by a repair in its current path, and sets as down the affected by a failure in its current path */
332                                Set<Route> routesFromDownToUp = currentNetPlan.getRoutesDown(wdmLayer);
333                                currentNetPlan.setLinksAndNodesFailureState(ev.linksToUp , ev.linksToDown , ev.nodesToUp , ev.nodesToDown);
334                                routesFromDownToUp.removeAll(currentNetPlan.getRoutesDown(wdmLayer));
335
336                                /* If something failed, and I am supposed to use protection or restoration... */
337                                if (isProtectionRecovery || isRestorationRecovery)
338                                {
339                                        /* Try to reroute the routes that are still failing */
340                                        for (Route r : new HashSet<Route> (currentNetPlan.getRoutesDown(wdmLayer)))
341                                        {
342                                                this.stat_numAttemptedToRecoverConnections ++;
343                                                this.stat_trafficAttemptedToRecoverConnections += r.getCarriedTrafficInNoFailureState();
344                                                final Demand cplDemand = this.cplWdm.getNodePairDemands(cplWdm.getNodeFromId(r.getIngressNode().getId()) , cplWdm.getNodeFromId(r.getEgressNode().getId()) , false).iterator().next();
345                                                
346                                                if (isProtectionRecovery)
347                                                {
348                                                        final Pair<WDMUtils.RSA,WDMUtils.RSA> originalRWA = wdmRouteOriginalRwa.get(r);
349                                                        final WDMUtils.RSA primaryRSA = originalRWA.getFirst();
350                                                        final WDMUtils.RSA backupRSA = originalRWA.getSecond();
351                                                        final boolean primaryUp = currentNetPlan.isUp (primaryRSA.seqLinks);
352                                                        final boolean backupUp = backupRSA == null? false : currentNetPlan.isUp (backupRSA.seqLinks);
353                                                        if (primaryUp) 
354                                                        {
355                                                                r.setSeqLinksAndProtectionSegments(primaryRSA.seqLinks);
356                                                                WDMUtils.setLightpathRSAAttributes(r , primaryRSA , false);
357                                                                this.stat_numSuccessfullyRecoveredConnections ++; 
358                                                                this.stat_trafficSuccessfullyRecoveredConnections += r.getCarriedTrafficInNoFailureState(); 
359                                                        }
360                                                        else if (backupUp) 
361                                                        {
362                                                                r.setSeqLinksAndProtectionSegments(Collections.singletonList(r.getPotentialBackupProtectionSegments().iterator().next()));
363                                                                WDMUtils.setLightpathRSAAttributes(r , backupRSA , false);
364                                                                this.stat_numSuccessfullyRecoveredConnections ++; 
365                                                                this.stat_trafficSuccessfullyRecoveredConnections += r.getCarriedTrafficInNoFailureState(); 
366                                                        }
367                                                }
368                                                else if (isRestorationRecovery)
369                                                {
370                                                        final Integer transponderTypeLp = transponderTypeOfNewLps.get(r);
371                                                        final double maxOpticalReachKm = transponderTypeLp == null? Double.MAX_VALUE : tpInfo.getOpticalReachKm(transponderTypeLp);
372                                                        final boolean isSignalRegenerationPossible = transponderTypeLp == null? true : tpInfo.isOpticalRegenerationPossible(transponderTypeLp); 
373                                                        final int numSlots = transponderTypeLp == null? new WDMUtils.RSA(r,false).getNumSlots() : tpInfo.getNumSlots(transponderTypeLp);
374                                                        WDMUtils.RSA rwa = computeValidPathNewRoute (cplDemand , currentNetPlan , numSlots , maxOpticalReachKm , isSignalRegenerationPossible);
375                                                        if (rwa != null)
376                                                        { 
377                                                                WDMUtils.releaseResources(new WDMUtils.RSA (r , false) , wavelengthFiberOccupancy, null);
378                                                                WDMUtils.allocateResources(rwa , wavelengthFiberOccupancy , null);
379                                                                r.setSeqLinksAndProtectionSegments(rwa.seqLinks);
380                                                                WDMUtils.setLightpathRSAAttributes(r , rwa , false);
381                                                                
382                                                                this.stat_numSuccessfullyRecoveredConnections ++; 
383                                                                this.stat_trafficSuccessfullyRecoveredConnections += r.getCarriedTrafficInNoFailureState(); 
384                                                        }
385                                                }
386                                        }
387                                }
388                                if (DEBUG) checkClashing (currentNetPlan); 
389                        } else if (event.getEventObject () instanceof SimEvent.DemandModify)
390                        {
391                                SimEvent.DemandModify ev = (SimEvent.DemandModify) event.getEventObject ();
392                                Demand d = ev.demand; 
393                                if (ev.modificationIsRelativeToCurrentOfferedTraffic) 
394                                        d.setOfferedTraffic(d.getOfferedTraffic() + ev.offeredTraffic);
395                                else
396                                        d.setOfferedTraffic(ev.offeredTraffic);
397                                if (DEBUG) checkClashing (currentNetPlan); 
398                        } else if (event.getEventObject () instanceof SimEvent.DemandRemove)
399                        {
400                                SimEvent.DemandRemove ev = (SimEvent.DemandRemove) event.getEventObject ();
401                                Demand d = ev.demand;
402                                for (Route lpToRemove : d.getRoutes())
403                                {
404                                        final Pair<WDMUtils.RSA,WDMUtils.RSA> originalRwa = wdmRouteOriginalRwa.get(lpToRemove);
405                                        if (originalRwa == null) return; // the ligtpath was already removed, because it was first accepted, but then blocked because of failures 
406                                        if (isRestorationRecovery) // restoration => the current route is the one to release
407                                                removeRoute_restoration(lpToRemove);
408                                        else
409                                                removeRoute_protection(lpToRemove , originalRwa);
410                                        transponderTypeOfNewLps.remove(lpToRemove);
411                                }
412                                d.remove();
413                                if (DEBUG) checkClashing (currentNetPlan); 
414                        }
415                        else if(event.getEventObject() instanceof WDMUtils.LightpathModify)
416                        {
417                                WDMUtils.LightpathModify modifyLpEvent = (WDMUtils.LightpathModify) event.getEventObject();
418                                final Route lpRoute = modifyLpEvent.lp;
419                                if (lpRoute != null) 
420                                {
421                                        WDMUtils.RSA oldRSA = new WDMUtils.RSA(lpRoute , false);
422                                        final Integer tpType = transponderTypeOfNewLps.get (lpRoute);
423                                        if (modifyLpEvent.rsa == null) throw new Net2PlanException ("Modifying the lightpath line rate requires setting the RSA also");
424                                        if (tpType == null) throw new Net2PlanException ("Cannot find the transponder type of the lightpath!");
425                                        if ((modifyLpEvent.carriedTraffic != 0) && (modifyLpEvent.carriedTraffic != tpInfo.getLineRateGbps(tpType))) throw new Net2PlanException ("Cannot change the line rate of an existing lightpath");
426                                        if ((modifyLpEvent.rsa.getNumSlots() != 0) && (modifyLpEvent.rsa.getNumSlots() != tpInfo.getNumSlots(tpType))) throw new Net2PlanException ("Cannot change the number of slots of an existing lightpath");
427                                        if ((modifyLpEvent.rsa.getLengthInKm() > tpInfo.getOpticalReachKm(tpType)) && (!tpInfo.isOpticalRegenerationPossible(tpType))) throw new Net2PlanException ("Cannot modify the lightpath RSA in sucha way that the lightpath length exceeds the transponder maximum reach");
428                                        
429                                        lpRoute.setCarriedTraffic(modifyLpEvent.carriedTraffic, modifyLpEvent.rsa.getNumSlots());
430                                        WDMUtils.releaseResources( oldRSA, wavelengthFiberOccupancy , null);
431                                        WDMUtils.allocateResources(modifyLpEvent.rsa , wavelengthFiberOccupancy , null);
432                                        lpRoute.setSeqLinksAndProtectionSegments(modifyLpEvent.rsa.seqLinks);
433                                        WDMUtils.setLightpathRSAAttributes(lpRoute , modifyLpEvent.rsa , false);
434                                }
435                        }
436                        else throw new Net2PlanException ("Unknown event type: " + event);
437                } catch (Exception e) {
438                        // TODO Auto-generated catch block
439                        e.printStackTrace();
440                        throw e;
441                }
442        }

444        public String finish(StringBuilder output, double simTime)
445        {
446                final double trafficBlockingOnConnectionSetup = stat_trafficOfferedConnections == 0? 0 : 1 - (stat_trafficCarriedConnections / stat_trafficOfferedConnections );
447                final double connectionBlockingOnConnectionSetup = stat_numOfferedConnections == 0? 0.0 : 1 - (((double) stat_numCarriedConnections) / ((double) stat_numOfferedConnections));
448                final double successProbabilityRecovery = stat_numAttemptedToRecoverConnections == 0? 0.0 : (((double) stat_numSuccessfullyRecoveredConnections) / ((double) stat_numAttemptedToRecoverConnections));
449                final double successProbabilityTrafficRecovery = stat_trafficAttemptedToRecoverConnections == 0? 0.0 : (((double) stat_trafficSuccessfullyRecoveredConnections) / ((double) stat_trafficAttemptedToRecoverConnections));
450                final double dataTime = simTime - stat_transitoryInitTime;
451                if (dataTime <= 0) { output.append ("<p>No time for data acquisition</p>"); return ""; }
452
453                output.append (String.format("<p>Total traffic of offered connections: number connections %d, total time average traffic: %f</p>", stat_numOfferedConnections, stat_trafficOfferedConnections / dataTime));
454                output.append (String.format("<p>Total traffic of carried connections: number connections %d, total time average traffic: %f</p>", stat_numCarriedConnections, stat_trafficCarriedConnections / dataTime));
455                output.append (String.format("<p>Blocking at connection setup: Probability of blocking a connection: %f, Fraction of blocked traffic: %f</p>", connectionBlockingOnConnectionSetup , trafficBlockingOnConnectionSetup));
456                output.append (String.format("<p>Number connections attempted to recover: %d (summing time average traffic: %f). </p>", stat_numAttemptedToRecoverConnections, stat_trafficAttemptedToRecoverConnections / dataTime));
457                output.append (String.format("<p>Number connections successfully recovered: %d (summing time average traffic: %f). </p>", stat_numSuccessfullyRecoveredConnections, stat_trafficSuccessfullyRecoveredConnections / dataTime));
458                output.append (String.format("<p>Probability of successfully recover a connection: %f. Proability weighted by the connection traffic: %f). </p>", successProbabilityRecovery, successProbabilityTrafficRecovery));
459//              output.append (String.format("<p>Total traffic carried at current state: %f. </p>", -1);
460                return "";
461        }

479        private WDMUtils.RSA computeValidPathNewRoute (Demand cplDemand , NetPlan currentNetPlan , int numContiguousSlots , double opticalReachKm , boolean isSignalRegenerationPossible)
480        {
481//              System.out.println ("computeValidPathNewRoute, demand: " + demand + " occupied: " + occupiedLinkCapacity);
482                /* If load sharing */
483                final double maxLightpathLengthKm = isSignalRegenerationPossible? Double.MAX_VALUE : opticalReachKm;
484                if (isLoadSharing)
485                {
486                        final ArrayList<Route> paths = new ArrayList<Route> (cplDemand.getRoutes());
487                        final int randomChosenIndex = rng.nextInt(paths.size ());
488                        final Route cplRoute = paths.get(randomChosenIndex);
489                        final List<Link> seqLinksThisNetPlan = cplToNetPlanLinks(cplRoute.getSeqLinksRealPath(), currentNetPlan);
490                        final Triple<Boolean,Integer,Double> rwaEval = computeValidWAFirstFit_path (seqLinksThisNetPlan , numContiguousSlots , maxLightpathLengthKm);
491                        if (rwaEval.getFirst()) return new WDMUtils.RSA (seqLinksThisNetPlan, rwaEval.getSecond() , numContiguousSlots , WDMUtils.computeRegeneratorPositions(seqLinksThisNetPlan , maxLightpathLengthKm)); else return null;
492                }
493                else if (isAlternateRouting || isLeastCongestedRouting)
494                {
495                        /* If alternate or LCR */
496                        WDMUtils.RSA lcrSoFar = null; double lcrIdleCapacitySoFar = -Double.MAX_VALUE; 
497                        for (Route cplRoute : cplDemand.getRoutes())
498                        {
499                                final List<Link> seqLinks = cplToNetPlanLinks(cplRoute.getSeqLinksRealPath() , currentNetPlan);
500                                final Triple<Boolean,Integer,Double> rwaEval = computeValidWAFirstFit_path (seqLinks , numContiguousSlots , maxLightpathLengthKm);
501//                              System.out.println ("Route: " + cplRoute + ", links : " + seqLinks + ", eval: " + rwaEval);
502                                if (!rwaEval.getFirst()) continue;
503                                if (isAlternateRouting) return new WDMUtils.RSA (seqLinks, rwaEval.getSecond(),numContiguousSlots , WDMUtils.computeRegeneratorPositions(seqLinks , maxLightpathLengthKm));
504                                if (rwaEval.getThird() > lcrIdleCapacitySoFar)
505                                {
506                                        lcrIdleCapacitySoFar = rwaEval.getThird();
507                                        lcrSoFar = new WDMUtils.RSA (seqLinks , rwaEval.getSecond());
508                                }
509                        }
510                        return (isAlternateRouting || (lcrSoFar == null))? (WDMUtils.RSA) null : lcrSoFar; 
511                }
512                else if (isSrgDisjointAwareLpRouting)
513                {
514                        final Node thisNpIngressNode = currentNetPlan.getNodeFromId (cplDemand.getIngressNode().getId ());
515                        final Node thisNpEgressNode = currentNetPlan.getNodeFromId (cplDemand.getEgressNode().getId ());
516                        WDMUtils.RSA chosenRWA = null; 
517                        Set<Route> existingRoutesBetweenPairOfNodes = currentNetPlan.getNodePairRoutes(thisNpIngressNode, thisNpEgressNode , false , wdmLayer);
518                        int cplChosenRoute_numSRGOverlappingRoutes = Integer.MAX_VALUE;
519                        for(Route cplCandidateRoute : cplDemand.getRoutes())
520                        {
521//                              if (forbiddenCplRoutes.contains (cplCandidateRoute)) continue;
522                                /* Check that the route has an available wavlength */
523                                final List<Link> seqLinks = cplToNetPlanLinks(cplCandidateRoute.getSeqLinksRealPath() , currentNetPlan);
524                                final Triple<Boolean,Integer,Double> rwaEval = computeValidWAFirstFit_path (seqLinks , numContiguousSlots , maxLightpathLengthKm);
525                                if (!rwaEval.getFirst()) continue;
526                                /* compute the number of SRG-route pairs this route overlaps with, respect to the already existing lightpaths  */
527                                final DoubleMatrix1D affectingSRGs = cplA_rs.viewRow (cplCandidateRoute.getIndex());
528                                int numOverlappingSRGs = 0; 
529                                for (Route currentRoute : existingRoutesBetweenPairOfNodes)
530                                        for (SharedRiskGroup srg : currentRoute.getSRGs()) if (affectingSRGs.get(srg.getIndex()) == 1) numOverlappingSRGs ++;
531                                if (numOverlappingSRGs < cplChosenRoute_numSRGOverlappingRoutes) 
532                                { 
533                                        cplChosenRoute_numSRGOverlappingRoutes = numOverlappingSRGs; 
534                                        chosenRWA = new WDMUtils.RSA (seqLinks , rwaEval.getSecond(),numContiguousSlots , WDMUtils.computeRegeneratorPositions(seqLinks, maxLightpathLengthKm));
535                                        if (numOverlappingSRGs == 0) break; // no need to search more
536                                }
537                        }
538                        return chosenRWA;
539                }
540                throw new RuntimeException ("Bad");
541        }

654        private Triple<Boolean,Integer,Double> computeValidWAFirstFit_path (List<Link> seqLinks , int numContiguousSlots , double maxOpticalReachKm)
655        {
656                if (seqLinks.isEmpty()) throw new RuntimeException ("Bad");
657                if (seqLinks.get(0).getOriginNode().isDown()) return Triple.of(false,-1,-1.0);
658                if (getLengthInKm(seqLinks) > maxOpticalReachKm) return Triple.of(false,-1,-1.0);
659                double worseCaseSpareCapacity = Double.MAX_VALUE; 
660                for (Link e : seqLinks) 
661                { 
662                        worseCaseSpareCapacity = Math.min(worseCaseSpareCapacity, e.getCapacity() - e.getOccupiedCapacityIncludingProtectionSegments());
663                        if (e.getDestinationNode().isDown() || e.isDown()) return Triple.of(false,-1,-1.0);
664                }
665                int wavelength = WDMUtils.spectrumAssignment_firstFit(seqLinks, wavelengthFiberOccupancy , numContiguousSlots);
666                if (wavelength != -1) 
667                        return Triple.of(true , wavelength , worseCaseSpareCapacity); 
668                else return Triple.of(false,-1,-1.0);
669        }

671        private Quadruple<Boolean,Integer,Integer,Double> computeValidWAFirstFit_pathPair (List<Link> seqLinks_1 , List<Link> seqLinks_2 , int numSlots , double maxOpticalReachKm)
672        {
673                if (seqLinks_1.isEmpty() || seqLinks_2.isEmpty()) throw new RuntimeException ("Bad");
674                if (seqLinks_1.get(0).getOriginNode().isDown()) return Quadruple.of(false,-1,-1,-1.0);
675                if (seqLinks_2.get(0).getOriginNode().isDown()) return Quadruple.of(false,-1,-1,-1.0);
676                if (getLengthInKm(seqLinks_1) > maxOpticalReachKm) return Quadruple.of(false,-1,-1,-1.0);
677                if (getLengthInKm(seqLinks_2) > maxOpticalReachKm) return Quadruple.of(false,-1,-1,-1.0);
678                double worseCaseSpareCapacity = Double.MAX_VALUE; 
679                for (Link e : seqLinks_1) 
680                { 
681                        worseCaseSpareCapacity = Math.min(worseCaseSpareCapacity, e.getCapacity() - e.getOccupiedCapacityIncludingProtectionSegments());
682                        if (e.getDestinationNode().isDown() || e.isDown()) return Quadruple.of(false,-1,-1,-1.0);
683                }
684                for (Link e : seqLinks_2) 
685                { 
686                        worseCaseSpareCapacity = Math.min(worseCaseSpareCapacity, e.getCapacity() - e.getOccupiedCapacityIncludingProtectionSegments());
687                        if (e.getDestinationNode().isDown() || e.isDown()) return Quadruple.of(false,-1,-1,-1.0);
688                }
689                Pair<Integer,Integer> seqWavelengths = WDMUtils.spectrumAssignment_firstFitTwoRoutes(seqLinks_1, seqLinks_2, wavelengthFiberOccupancy,numSlots);
690                if (seqWavelengths != null) 
691                        return Quadruple.of(true , seqWavelengths.getFirst () , seqWavelengths.getSecond() , worseCaseSpareCapacity); 
692                else return Quadruple.of(false,-1,-1,-1.0);
693        }

734        private void checkDisjointness (List<Link> p1 , List<Link> p2 , int disjointnessType)
735        {
736                if ((p1 == null) || (p2 == null)) throw new RuntimeException ("Bad");
737                if (disjointnessType == 0) // SRG disjoint
738                {
739                        Set<SharedRiskGroup> srg1 = new HashSet<SharedRiskGroup> (); for (Link e : p1) srg1.addAll (e.getSRGs());
740                        for (Link e : p2) for (SharedRiskGroup srg : e.getSRGs()) if (srg1.contains(srg)) throw new RuntimeException ("Bad");
741                }
742                else if (disjointnessType == 1) // link and node
743                {
744                        Set<NetworkElement> resourcesP1 = new HashSet<NetworkElement> (); boolean firstLink = true; 
745                        for (Link e : p1) { resourcesP1.add (e); if (firstLink) firstLink = false; else resourcesP1.add (e.getOriginNode()); }
746                        for (Link e : p2) if (resourcesP1.contains(e)) throw new RuntimeException ("Bad");
747                }
748                else if (disjointnessType == 2) // link 
749                {
750                        Set<Link> linksP1 = new HashSet<Link> (p1); 
751                        for (Link e : p2) if (linksP1.contains (e)) throw new RuntimeException ("Bad: p1: " + p1 + ", p2: " + p2);
752                }
753                else throw new RuntimeException ("Bad");
754                        
755        }
